       ```python
       def _build_field_map(model_class: type[BaseModel]) -> dict[str, str]:
           """Build {arrow_column_name: pydantic_field_name} mapping.
           Resolution priority (ALIAS-01): validation_alias > alias > field_name.
           """
           config = model_class.model_config
           # ALIAS-03: Reject AliasGenerator on model config
           alias_gen = config.get("alias_generator")
           if alias_gen is not None:
               raise NotImplementedError(
                   f"AliasGenerator on {model_class.__name__} is not supported. "
                   "Use explicit per-field aliases instead."
               )
           field_map: dict[str, str] = {}
           for field_name, field_info in model_class.model_fields.items():
               va = field_info.validation_alias
               if va is not None:
                   if isinstance(va, (AliasPath, AliasChoices)):
                       raise NotImplementedError(
                           f"Field {field_name!r} uses {type(va).__name__} as "
                           "validation_alias, which is not supported."
                       )
                   lookup_name = va  # str
               elif field_info.alias is not None:
                   lookup_name = field_info.alias
               else:
                   lookup_name = field_name
               field_map[lookup_name] = field_name
           # ALIAS-02: populate_by_name / validate_by_name support
           accept_by_name = (
               config.get("validate_by_name", False)
               or config.get("populate_by_name", False)
           )
           if accept_by_name:
               for field_name in model_class.model_fields:
                   if field_name not in field_map:
                       field_map[field_name] = field_name
           return field_map
       ```

       ```python
       def _resolve_columns(self, schema: pa.Schema) -> tuple[list[int], list[str]]:
           col_indices: list[int] = []
           field_names: list[str] = []
           missing: list[str] = []
           for lookup_name, field_name in self._field_map.items():
               col_idx = schema.get_field_index(lookup_name)
               if col_idx < 0:
                   field_info = self._model_class.model_fields[field_name]
                   if field_info.is_required():
                       missing.append(lookup_name)
                   continue
               col_indices.append(col_idx)
               field_names.append(field_name)
           if missing:
               raise ValueError(
                   f"Arrow schema is missing required columns: {missing}. "
                   f"Available columns: {schema.names}"
               )
           return col_indices, field_names
       ```

       ```python
       class TestAliasResolution:
           def test_validation_alias_priority(self) -> None:
               batch = pa.record_batch({"userId": [1, 2], "displayName": ["a", "b"]})
               results = ArrowModelConverter(ValidationAliasModel).convert(batch)
               assert len(results) == 2
               assert results[0].user_id == 1
               assert results[0].display_name == "a"

           def test_alias_fallback(self) -> None:
               batch = pa.record_batch({"userId": [1, 2], "displayName": ["a", "b"]})
               results = ArrowModelConverter(AliasModel).convert(batch)
               assert results[0].user_id == 1
               assert results[0].display_name == "a"

           def test_field_name_fallback(self) -> None:
               batch = pa.record_batch({"id": [1], "name": ["x"], "score": [1.0], "active": [True]})
               results = ArrowModelConverter(MixedModel).convert(batch)
               assert results[0].id == 1

           def test_mixed_alias_types(self) -> None:
               batch = pa.record_batch({"userId": [1], "displayName": ["a"], "email": ["x@y.z"]})
               results = ArrowModelConverter(MixedAliasModel).convert(batch)
               assert results[0].user_id == 1
               assert results[0].display_name == "a"
               assert results[0].email == "x@y.z"

           def test_populate_by_name(self) -> None:
       ```

       ```python
       class TestSchemaValidation:
           def test_missing_required_field_raises(self) -> None:
               batch = pa.record_batch({"id": [1, 2]})
               converter = ArrowModelConverter(MixedModel)
               with pytest.raises(ValueError, match="missing required columns"):

           def test_missing_multiple_fields_lists_all(self) -> None:
               batch = pa.record_batch({"wrong": [1, 2]})
               converter = ArrowModelConverter(MixedModel)
               with pytest.raises(ValueError, match="missing required columns") as exc_info:
                   converter.convert(batch)
               error_msg = str(exc_info.value)
               # All 4 fields of MixedModel should be listed as missing

           def test_optional_field_missing_uses_default(self) -> None:
               # OptionalFieldModel has required "id", optional "name" (default "default_name"), optional "score" (default None)

               assert results[0].name == "default_name"

               assert results[0].id == 1
       ```
