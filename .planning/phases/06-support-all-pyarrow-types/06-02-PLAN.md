<objective>
Add ColumnExtractor support for interval types, container types (FixedSizeList, Map), RunEndEncoded pre-unpacking, and Union (sparse + dense) arrays.

Purpose: Completes the full pyarrow type coverage. Interval types handle Arrow-specific duration concepts. FixedSizeList and Map are container types requiring recursive extraction. RunEndEncoded uses the existing Dictionary pre-unpack pattern. Union is the most complex -- each row can hold a different type.

Output: Updated extract.rs with 7 new ColumnExtractor variants (3 intervals + FixedSizeList + Map + Union + ListView), updated lib.rs with REE unpack support, comprehensive tests.
</objective>

fn unpack_columns(
    columns: &[ArrayRef],
    schema: &arrow_schema::SchemaRef,
    col_indices: &[usize],
) -> Result<Vec<ArrayRef>, PyErr> {
    col_indices.iter().map(|&idx| {
        let col = &columns[idx];
        let dt = schema.field(idx).data_type();
        match dt {
            DataType::Dictionary(_, value_type) => {
                arrow_cast::cast(col.as_ref(), value_type.as_ref()).map_err(...)
            }
            // ADD: DataType::RunEndEncoded(_, value_field) => { ... }
            _ => Ok(col.clone()),
        }
    }).collect()
}

       let type_map: Vec<(i8, DataType)> = union_fields

           .map(|(tid, field)| (tid, field.data_type().clone()))
           .collect();
       Ok(ColumnExtractor::Union(arr, type_map))
   }

   **FixedSizeList:** (identical pattern to List)
   ```rust
   ColumnExtractor::FixedSizeList(arr, child_dt) => {
       if arr.is_null(row) {
           Ok(py.None())
       } else {
           let child_array = arr.value(row);
           let len = child_array.len();
           let child_ext = prepare_extractor(py, child_array.as_ref(), child_dt, None)?;
           let mut items: Vec<PyObject> = Vec::with_capacity(len);
           for j in 0..len {
               items.push(child_ext.extract_value(py, j)?);
           }
           Ok(PyList::new(py, &items)?.into_any().unbind())
       }
   }
   ```

   **Map:**
   ```rust
   ColumnExtractor::Map(arr, key_dt, val_dt) => {
       if arr.is_null(row) {
           Ok(py.None())
       } else {
           let entries = arr.value(row); // StructArray with keys+values children
           let keys_arr = entries.column(0);
           let vals_arr = entries.column(1);
           let key_ext = prepare_extractor(py, keys_arr.as_ref(), key_dt, None)?;
           let val_ext = prepare_extractor(py, vals_arr.as_ref(), val_dt, None)?;
           let len = entries.len();
           let mut items: Vec<PyObject> = Vec::with_capacity(len);
           for j in 0..len {
               let k = key_ext.extract_value(py, j)?;
               let v = val_ext.extract_value(py, j)?;
               let pair = PyTuple::new(py, &[k, v])?;
               items.push(pair.into_any().unbind());
           }
           Ok(PyList::new(py, &items)?.into_any().unbind())
       }
   }
   ```

   **Union:**
   ```rust
   ColumnExtractor::Union(arr, type_map) => {
       if arr.is_null(row) {
           Ok(py.None())
       } else {
           let tid = arr.type_id(row);
           let child = arr.child(tid);
           let child_idx = match arr.offsets() {
               Some(_) => arr.value_offset(row) as usize, // Dense
               None => row,                                 // Sparse
           };
           let child_dt = type_map.iter()
               .find(|(id, _)| *id == tid)
               .map(|(_, dt)| dt)
               .ok_or_else(|| PyErr::new::<pyo3::exceptions::PyTypeError, _>(
                   format!("Unknown union type_id: {tid}")
               ))?;
           let child_ext = prepare_extractor(py, child.as_ref(), child_dt, None)?;
           child_ext.extract_value(py, child_idx)
       }
   }
   ```
